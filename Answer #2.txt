#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int find(int a[][3])
{
	int i,j,c=0;
	for(i=1;i<2;i++)
	{
		for(j=0;j<3;j++)
		{
			if(a[i+1][j]>a[i][j])   //checking i+1 is better than i or not if yes returns 1 else returns 0
			c=1;
			else
			continue;
		}
	}
	return c;
}
int main()
{
	int t;
	cin>>t;// number of test cases
	while(t--)
	{
		int i,j,a[3][3];
		for(i=0;i<3;i++)
		for(j=0;j<3;j++)
		cin>>a[i][j];
		int r=find(a);//code check student is better or not
		if(r)
		cout<<"yes"<<endl;
		else
		cout<<"no"<<endl;
	}
	return 0;
}
